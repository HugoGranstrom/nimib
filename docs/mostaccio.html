Example of usage of mustache

From nim-mustache.

Usage

Step 1.

import
  mustache, tables

Step 2.

var c = newContext()
c["i"] = 1
c["f"] = 1.0
c["s"] = "hello world"
c["a"] = @[{"k": "v"}.toTable]
c["t"] = {"k": "v"}.toTable
c["l"] = proc (s: string; c: Context): string =
  "&lt;b&gt;" & s.render(c) & "&lt;/b&gt;"

Step 3.

let s = """{{i}} {{f}} {{s}}
{{#a}}
  {{k}}
{{/a}}

{{#t}}
  {{k}}
{{/t}}

{{#l}}
  {{s}}
{{/l}}
"""
echo(s.render(c))
